Indexer des PDFs :

python rag_colpali_miniproject.py index \
  --pdf-dir ./data/pdfs \
  --img-dir ./data/images \
  --index-dir ./data/index


Rechercher (top-k pages) :

python rag_colpali_miniproject.py query \
  --index-dir ./data/index \
  --img-dir ./data/images \
  --question "Quels sont les symptômes de la phénylcétonurie ?" \
  --k 5
